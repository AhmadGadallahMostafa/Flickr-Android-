36       @Override
37       public void onClick(View v) {
38           Intent intent;
39           switch(v.getId()){
40               case(R.id.favButtonPostImg):
41                   intent = new Intent(this,FavouritesCommentsActivity.class );
42                   intent.putExtra("tabLayout","0");
43                   intent.putExtra("post",postOfImage);
44                   this.startActivity(intent);
45                   break;
46               case(R.id.shareButtonPostImg):
47                   intent = new Intent(Intent.ACTION_SEND );
48                   intent.setType("text/plain");
49                   intent.putExtra(Intent.EXTRA_SUBJECT,"Link: ");
50                   intent.putExtra(Intent.EXTRA_TEXT,"http://www.thealphaflickr.xyz/photo/"+postOfImage.getPostId());
51                   this.startActivity(Intent.createChooser(intent,"Share via: "));
52                   break;
53               case(R.id.commentsButtonPostImg):
54                   intent = new Intent(this,FavouritesCommentsActivity.class );
55                   intent.putExtra("tabLayout","1");
56                   intent.putExtra("post",postOfImage);
57                   this.startActivity(intent);
58                   break;
59               case(R.id.postImageLayout):
60               case(R.id.imageViewPostImg):
61                   if(isVisible){
62                       topToolbar.setVisibility(View.GONE);
63                       bottomToolbar.setVisibility(View.GONE);
64                       captionToolbar.setVisibility(View.GONE);
65                       isVisible=false;
66                   }else{
67                       topToolbar.setVisibility(View.VISIBLE);
68                       bottomToolbar.setVisibility(View.VISIBLE);
69                       captionToolbar.setVisibility(View.VISIBLE);
70                       isVisible=true;
71                   }
72                   break;
73               default:
74                   break;
75           }
76       }

98           favButton.setOnClickListener(this);
99           commentButton.setOnClickListener(this);
100          shareButton.setOnClickListener(this);
101          constraintLayout.setOnClickListener(this);
102          imageViewPost.setOnClickListener(this);
103          userNameText.setText(postOfImage.getPostUserProfile().getName());
104          Glide.with(this).asBitmap().load(postOfImage.getPostUserProfile().getProfilePicURL()).into(profImg);
105          Glide.with(this).asBitmap().load(postOfImage.getImageURL()).into(imageViewPost);
106          String cap = postOfImage.getCaption();
107          if (cap == null) cap = "";
108          captionText.setText(cap);
