96       private void getPhotos(){
97           String url ="https://api.unsplash.com/photos?client_id=lw8JVwKlDWjEhUxqdnB2tRel7Fduqc2Z1_DdXyAzNzI&page="+ pageNum;

102              @Override
103              public void onResponse(String response) {
104                  JSONArray jsonArray;
105  
106                  try{
107                      jsonArray = new JSONArray(response);
108                      //posts = new ArrayList<>();
109                      for(int i=0;i<10;i++){
110                          JSONObject jsonObject = jsonArray.getJSONObject(i);
111                          String idImg=jsonObject.get("id").toString();
112                          System.out.println("id:"+idImg);
113                          UserProfile userProfile = new UserProfile();
114                          String userName = jsonObject.getJSONObject("user").get("name").toString();
115                          System.out.println("username:"+userName);
116                          userProfile.setName(userName);
117                          String urlProfImg =jsonObject.getJSONObject("user").getJSONObject("profile_image").get("small").toString();
118                          userProfile.setProfilePicURL(urlProfImg);
119                          String urlImg =jsonObject.getJSONObject("urls").get("raw").toString();
120                          System.out.println("url:"+urlImg);
121                          Post postPointer = new Post(idImg,userProfile,urlImg,i);
122                          String caption = jsonObject.get("alt_description").toString();
123                          postPointer.setCaption(caption);
124                          String date = jsonObject.get("created_at").toString();
125                          String daten = date.substring(0,19);
126                          daten = daten + ".0" + date.substring(19,25);
127                          SimpleDateFormat d = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSXXX");
128                          Date datePosted = d.parse(daten);
129                          postPointer.setDatePosted(datePosted);
130                          int favs = Integer.parseInt(jsonObject.get("likes").toString());
131                          postPointer.setFavouriteCount(favs);
132                          posts.add(postPointer);
133  
134                          postAdapter.setPosts(posts);
135  
136                      }
137                      //pageNum++;
